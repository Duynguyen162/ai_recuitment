kích hoạt môi trường ảo:
    python -m venv venv
    venv\Scripts\activate

chạy lệnh cài đặt các thư viện:
    python -m pip install -r requirements.txt

chạy dự án :
    uvicorn app.main:app --reload

đẩy bảng mới tạo xuống db:
    #Tạo file migration mới dựa trên sự khác biệt Model SQLAlchemy (code Python) vs Database hiện tại
        alembic revision --autogenerate -m "create profile and company"
    # Chạy các migration chưa áp dụng và cập nhật database.
        alembic upgrade head  


tải thư viện mã hóa và tạo JWT
    pip install "passlib[bcrypt]" python-jose python-multipart
